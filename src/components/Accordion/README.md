# Accordion

Component to show information in an expandable panel.

## Usage

```tsx
import { Accordion } from '@totase/mono-ui';

function Component() {
  return (
    <Accordion>
      <Accordion.Header>Accordion header</Accordion.Header>
      <Accordion.Content>Expandable content</Accordion.Content>
    </Accordion>
  );
}
```
